    let currentQuestionIndex = 0;
    let score = 0;
    let answeredQuestions = 0;
    let userAnswers = new Array(questions.length).fill(null);
    
    // Catégories pour l'analyse des résultats
    const categories = {
        "Transcription": { correct: 0, total: 0 },
        "Code génétique": { correct: 0, total: 0 },

    function displayQuestion(index) {
        const question = questions[index];
        const quizContainer = document.getElementById('quiz-container');
        
        // Compter les questions par catégorie
        categories[question.title].total++;
        
        // Créer l'élément de question
        const questionElement = document.createElement('div');
        questionElement.id = `question-${index}`;
        questionElement.className = 'question-card bg-white p-6 rounded-xl shadow-md';
        questionElement.style.display = index === currentQuestionIndex ? 'block' : 'none';
        
        // Contenu de la question
        questionElement.innerHTML = `
            <div class="flex justify-between items-center mb-4">
                <span class="bg-indigo-100 text-indigo-800 text-xs font-semibold px-3 py-1 rounded-full">${question.title}</span>
                <span class="text-gray-500 text-sm">Question ${index + 1}/50</span>
            </div>
            <h3 class="text-lg font-semibold mb-4">${question.question}</h3>
            <div class="space-y-3 mb-4">
                ${question.options.map((option, i) => `
                    <div class="option border border-gray-300 rounded-lg p-3 cursor-pointer flex items-center" data-index="${i}">
                        <span class="text-indigo-700 font-bold mr-3">${String.fromCharCode(65 + i)}</span>
                        <span>${option}</span>
                    </div>
                `).join('')}
            </div>
            <div class="explanation mt-4 p-4 bg-indigo-50 rounded-lg">
                <p class="font-semibold mb-1">Explication:</p>
                <p>${question.explanation}</p>
            </div>
        `;
        
        // Ajouter la question au conteneur
        if (document.getElementById(`question-${index}`)) {
            document.getElementById(`question-${index}`).replaceWith(questionElement);
        } else {
            quizContainer.appendChild(questionElement);
        }
        
        // Ajouter les écouteurs d'événements aux options
        const options = questionElement.querySelectorAll('.option');
        options.forEach(option => {
            option.addEventListener('click', function() {
                // Désélectionner toutes les options
                options.forEach(opt => opt.classList.remove('selected'));
                // Sélectionner l'option cliquée
                this.classList.add('selected');
                // Enregistrer la réponse de l'utilisateur
                userAnswers[index] = parseInt(this.dataset.index);
            });
            
            // Restaurer la sélection précédente si elle existe
            if (userAnswers[index] !== null && parseInt(option.dataset.index) === userAnswers[index]) {
                option.classList.add('selected');
            }
        });
        
        // Mettre à jour la barre de progression
        updateProgressBar();
    }

    function checkAnswer() {
        const currentQuestion = questions[currentQuestionIndex];
        const selectedOption = document.querySelector(`#question-${currentQuestionIndex} .option.selected`);
        
        if (!selectedOption) {
            alert("Veuillez sélectionner une réponse avant de vérifier.");
            return;
        }
        
        const selectedIndex = parseInt(selectedOption.dataset.index);
        const isCorrect = selectedIndex === currentQuestion.correctAnswer;
        
        // Mettre à jour le score
        if (isCorrect) {
            score++;
            categories[currentQuestion.title].correct++;
        }
        
        // Marquer les options comme correctes ou incorrectes
        const options = document.querySelectorAll(`#question-${currentQuestionIndex} .option`);
        options.forEach((option, index) => {
            option.classList.remove('selected');
            if (index === currentQuestion.correctAnswer) {
                option.classList.add('correct');
            } else if (index === selectedIndex) {
                option.classList.add('incorrect');
            }
        });
        
        // Afficher l'explication
        document.querySelector(`#question-${currentQuestionIndex} .explanation`).style.display = 'block';
        
        // Désactiver les options
        options.forEach(option => {
            option.style.pointerEvents = 'none';
        });
        
        // Mettre à jour le nombre de questions répondues
        answeredQuestions++;
        
        // Mettre à jour le score affiché
        document.getElementById('score-text').textContent = `Score: ${score}/${answeredQuestions}`;
        
        // Activer le bouton suivant
        document.getElementById('check-btn').style.display = 'none';
        document.getElementById('next-btn').style.display = 'block';
    }

    function nextQuestion() {
        // Cacher la question actuelle
        document.getElementById(`question-${currentQuestionIndex}`).style.display = 'none';
        
        // Passer à la question suivante
        currentQuestionIndex++;
        
        // Vérifier si c'est la fin du quiz
        if (currentQuestionIndex >= questions.length) {
            showResults();
            return;
        }
        
        // Afficher la nouvelle question
        displayQuestion(currentQuestionIndex);
        
        // Réinitialiser les boutons
        document.getElementById('check-btn').style.display = 'block';
        document.getElementById('next-btn').style.display = 'none';
        
        // Activer/désactiver les boutons de navigation
        document.getElementById('prev-btn').disabled = currentQuestionIndex === 0;
    }

    function prevQuestion() {
        // Cacher la question actuelle
        document.getElementById(`question-${currentQuestionIndex}`).style.display = 'none';
        
        // Passer à la question précédente
        currentQuestionIndex--;
        
        // Afficher la nouvelle question
        displayQuestion(currentQuestionIndex);
        
        // Activer/désactiver les boutons de navigation
        document.getElementById('prev-btn').disabled = currentQuestionIndex === 0;
    }

    function updateProgressBar() {
        const progressPercentage = (currentQuestionIndex / questions.length) * 100;
        document.getElementById('progress-bar').style.width = `${progressPercentage}%`;
        document.getElementById('progress-text').textContent = `Question ${currentQuestionIndex + 1}/${questions.length}`;
    }

            const percentage = (correct / total) * 100;
            
            const listItem = document.createElement('li');
            listItem.textContent = `${category}: ${correct}/${total} (${percentage.toFixed(1)}%)`;
            
            if (percentage >= 60) {
                strengths.appendChild(listItem);
            } else {
                weaknesses.appendChild(listItem);
            }
        }
        
        // Si aucune force ou faiblesse n'est identifiée
        if (strengths.children.length === 0) {
            strengths.innerHTML = '<li>Aucun point fort identifié. Continuez à travailler!</li>';
        }
        if (weaknesses.children.length === 0) {
            weaknesses.innerHTML = '<li>Félicitations! Aucune faiblesse majeure identifiée.</li>';
        }
    }

    function restartQuiz() {
        // Réinitialiser les variables
        currentQuestionIndex = 0;
        score = 0;
        answeredQuestions = 0;
        userAnswers = new Array(questions.length).fill(null);
        
        for (const category in categories) {
            categories[category].correct = 0;
            categories[category].total = 0;
        }
        
        // Vider le conteneur de quiz
        document.getElementById('quiz-container').innerHTML = '';
        document.getElementById('quiz-container').style.display = 'block';
        
        // Réinitialiser les boutons

        document.getElementById('check-btn').style.display = 'block';
        document.getElementById('next-btn').style.display = 'none';
        document.getElementById('prev-btn').disabled = true;
        
        // Cacher les résultats
        document.getElementById('results').style.display = 'none';
        
        // Réinitialiser le score affiché
        document.getElementById('score-text').textContent = 'Score: 0/0';
        
        // Afficher la première question
        displayQuestion(0);
    }
